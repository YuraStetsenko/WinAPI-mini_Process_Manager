// Отримання та додавання списку процесів до ListView
    std::vector<std::wstring> processList = GetProcessList();
    for (const auto& process : processList) {
        LVITEM lvItem;
        lvItem.mask = LVIF_TEXT;
        lvColumn.cx = 300;
        lvItem.iItem = ListView_GetItemCount(hwndListView); // Індекс рядка
        lvItem.iSubItem = 0; // Індекс стовпця
        lvItem.pszText = const_cast<wchar_t*>(process.c_str());
        ListView_InsertItem(hwndListView, &lvItem);
    }

    
    
    #define IDC_LISTVIEW 100

    // Ініціалізація ListView
    INITCOMMONCONTROLSEX icex;
    icex.dwSize = sizeof(INITCOMMONCONTROLSEX);
    icex.dwICC = ICC_LISTVIEW_CLASSES;
    InitCommonControlsEx(&icex);

    HWND hwndListView = CreateWindowEx(0, WC_LISTVIEW, L"",
        WS_VISIBLE | WS_CHILD | LVS_REPORT | LVS_SINGLESEL,
        10, 10, 600, 400,
        hwnd, (HMENU)IDC_LISTVIEW, hInstance, NULL);
